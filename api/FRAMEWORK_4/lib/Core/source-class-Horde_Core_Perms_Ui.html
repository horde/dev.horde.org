 94:                 'class' => 'treeHdrSpacer'

213:                     if (strpos($name, $prefix) === 0) {
214:                         unset($child_perms[substr($name, $length)]);
215:                     }
216:                 }
217:                 $this->_form->addVariable(Horde_Core_Translation::t("Permissions"), 'child', 'enum', true, false, null, array($child_perms));
