 73:                 if ($op == 'AND') {

 90:                             break;
 91:                         }
 92:                     }
 93:                 } else {
 94:                     $found = false;
 95:                 }
 96:             }

113:     protected function _match($contact, $val)
114:     {
115:         if (!isset($contact[$val['field']])) {
116:             return false;

118:         switch ($val['op']) {
