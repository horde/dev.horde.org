 50:      *
 51:      * @param string|array $args  List of arguments to pass to compose.php.

259:                     $new_filter[] = IMP_Sentmail::MDN;
260:                     break;

264:                     break;
265: 
266:                 case 'redirect':

268:                     break;

274:             }
