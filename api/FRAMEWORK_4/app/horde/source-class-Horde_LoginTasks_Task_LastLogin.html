61:             $ptrdname = $host;
62:             try {
63:                 if ($response = $dns->query($host, 'PTR')) {
64:                     foreach ($response->answer as $val) {
65:                         if (isset($val->ptrdname)) {

67:                             break;
68:                         }
69:                     }
70:                 }
