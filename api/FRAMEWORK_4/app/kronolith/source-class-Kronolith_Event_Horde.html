132:         }
133:         $this->_params = $event['params'];
134:         $this->_link = !empty($event['link']) ? $event['link'] : null;
135:         $this->url = !empty($event['url']) ? (string)$event['url'] : null;
136:         $this->_editLink = !empty($event['edit_link']) ? $event['edit_link'] : null;
