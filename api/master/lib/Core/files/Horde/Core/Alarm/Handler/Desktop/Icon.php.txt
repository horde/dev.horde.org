 */
class Horde_Core_Alarm_Handler_Desktop_Icon
{
    /**
     * Icon path.
     *
     * @var string
     */
    protected $_path;

    /**
     * Constructor.
     *
     * @param string $path  Icon path.
     */
    public function __construct($path)
    {
        $this->_path = $path;
    }

    /**
     */
    public function __toString()
    {
        return strval(Horde_Themes::img($this->_path)->fulluri);
    }

}
