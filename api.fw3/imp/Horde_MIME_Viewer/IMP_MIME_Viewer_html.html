<img[^>]+?src=
# <input> tags
|<input[^>]+?src=

|<body[^>]+?background=|<td[^>]*background=|<table[^>]*background=
# "style" attributes; match 2; quotes: match 3
|(style=\s*("|\')?[^>]*?background(?:-image)?:(?(3)[^"\']|[^>])*?url\s*\()

\s*
